$PROBLEM 5-FU PK model (1 compartiment) focei
;; 1. Based on: 5fu_pop_focei
;; 2. Description: 5-FU PK model (1 compartiment)
;; x1. Author: Olga
;; 3. Label: 5-FU PK model

; FOCE-I method, Data augmentation


$INPUT ID	TIME	REGIME	CYC	MDV	DV	EVID	AGE	WT	HGT	SEX	BSA	AMT	RATE	LBM	FM

$DATA DataAugmentation_dataset_split8.csv IGNORE=#  ; augmentation template

$SUBROUTINES ADVAN1 TRANS2

$PK
CLBSA= (1+ THETA(4) * (BSA-1.97)) ;BSA as covariate on clearance, BSA centered on training population median (split 8)
CLCOV= CLBSA

;Structural model
TVCL=CLCOV*THETA(1)
CL=TVCL*EXP(ETA(1))
TVV=THETA(2)
V=TVV*EXP(ETA(2))

S1=V
AUC=AMT/CL

$ERROR
IPRED=F
DEL=0
W=SQRT((THETA(3)*IPRED)**2) ; proportional error model
Y=IPRED+W*EPS(1) 
IF(W.EQ.0) DEL=0.0001
IRES= DV*IPRED
IWRES= IRES/(W+DEL)

$THETA ; parameter estimates from FOCE-I estimation
(0,225); CL; Wilhelm et al.
46.1 FIX ; V; Wilhelm et al.
(0,0.217) ; prop. error, residual variability from Wilhelm et al.
0.818 ; CLBSA; Wilhelm  et al.

$OMEGA
0.0588; IIV CL; Wilhelm et al.
0.511 FIX ; IIV V; Wilhelm et al.

$SIGMA 1 FIX

$SIMULATION (12345) (54321) ONLYSIM SUBPROBLEMS=1000
$TABLE ID AMT TIME REGIME RATE CYC MDV EVID AGE WT HGT SEX BSA LBM FM IPRED DV NOAPPEND ONEHEADER NOPRINT
